 22class CrestMetadata:
 23    """
 24    The databases that `crest4` needs are slightly too large to be distributed
 25    with the source tarball on PyPI. Therefore, we host them on a separate
 26    server. Currently, this is S3 on AWS.
 27
 28    Over the course of years however these servers might be shutdown.
 29    To avoid this inconvenience, instead of hard-coding the server address here
 30    in the source file, we instead maintain a metadata file.
 31
 32    This metadata file contains the up-to-date URLs of the databases
 33    to download. It is placed in the GitHub repository and only the location
 34    of the metadata file is hardcoded into the source code.
 35
 36    This enables updating the URLs without having to issue a new release of
 37    `crest4`.
 38
 39    A good read on this subject is the following:
 40
 41    https://www.dampfkraft.com/code/distributing-large-files-with-pypi.html
 42    """
 43
 44    # The location of the metadata file is hardcoded #
 45    base_url  = "https://raw.githubusercontent.com/xapple/"
 46    repo_path = "crest4/master/crest4/crest4_db_urls.json"
 47
 48    # This will always retrieve the raw content of the latest version #
 49    metadata_url = base_url + repo_path
 50
 51    # To see the HTML version on GitHub simply head to this address instead:
 52    # https://github.com/xapple/crest4/blob/master/crest4/crest4_db_urls.json
 53
 54    @property_cached
 55    def db_urls(self):
 56        """
 57        Once the `Metadata` object is created simply access it like this:
 58        >>> print(metadata.db_urls['silvamod138']['url'])
 59        """
 60        # Download the gist #
 61        content = retrieve_from_url(self.metadata_url)
 62        # Parse it #
 63        data = json.loads(content)
 64        # Validate it #
 65        assert list(data) == ['crest4_databases']
 66        # Return it #
 67        return data['crest4_databases']
 68
 69# The metadata should be very rarely updated -- we create a singleton #
 70metadata = CrestMetadata()

 73class CrestDatabase:
 74    """
 75    This object represents a database of taxonomically curated DNA sequences
 76    that was subsequently specifically formatted for use with `crest4`.
 77    It comes with a FASTA file, as well as a tree.
 78    """
 79
 80    # Default location for a database #
 81    default_dir = os.path.expanduser('~/.crest4/')
 82
 83    # The environment variable that the user can set #
 84    environ_var = "CREST4_DIR"
 85
 86    def __init__(self, short_name, long_name, base_dir=None):
 87        """
 88        The user can specify a path pointing to a directory that contains all
 89        the required files, and having the same prefix as the directory name.
 90        It must include the database FASTA file as well as the `.map` and
 91        `.names` file.
 92        """
 93        # Base required attributes #
 94        self.short_name = short_name
 95        self.long_name = long_name
 96        # If the user specifies a base_dir, then no need to download #
 97        if base_dir is not None:
 98            self.downloaded = True
 99        # If the user doesn't specify a base_dir path, then we should
100        # determine where the actual files will be located on disk and
101        # check if the user has set the special environment variable.
102        if base_dir is None:
103            base_dir = os.environ.get(self.environ_var, self.default_dir)
104        # Convert to a DirectoryPath #
105        self.base_dir = DirectoryPath(base_dir)
106        # And make it absolute #
107        self.base_dir = DirectoryPath(self.base_dir.absolute_path)
108
109    def __repr__(self):
110        """A simple representation of this object to avoid memory addresses."""
111        return "<%s object at '%s'>" % (self.__class__.__name__, self.path)
112
113    @property
114    def tag(self):
115        """
116        Some methods will look for the `tag` attribute to print the name of the
117        database.
118        """
119        return self.short_name
120
121    @property
122    def tarball(self):
123        """Determine where the database `.tar.gz` will be located on disk."""
124        return self.base_dir + self.short_name + '.tar.gz'
125
126    @property
127    def path(self):
128        """The path to the FASTA file."""
129        return self.base_dir + self.short_name + '/' + \
130               self.short_name + '.fasta'
131
132    @property_cached
133    def downloaded(self):
134        """
135        Determine if the database has been downloaded to disk and uncompressed
136        previously. Returns `True` or `False`.
137        """
138        # Check if the tree is there and not empty #
139        return bool(self.path.replace_extension('tre'))
140
141    @property
142    def url(self):
143        """
144        Retrieve the URL of the database file to download by first downloading
145        a metadata file (see the `Metadata` class above).
146        """
147        # Check that the key exists #
148        if self.short_name not in metadata.db_urls:
149            msg = "The database '%s' is not in the list of downloadable" \
150                  " databases. Please update the file `crest4_db_urls.json`" \
151                  " to include it."
152            raise Exception(msg % self.short_name)
153        # Get the URL #
154        return metadata.db_urls[self.short_name]['url']
155
156    def download(self):
157        """
158        Download the database file, uncompress it, and save it to disk.
159        """
160        # Compose a message so the user knows why it's taking time #
161        message = "The database '%s' has not been downloaded yet." \
162                  " This process will start now and might take some time" \
163                  " depending on your internet connection. Please be" \
164                  " patient. The result will be saved to '%s'. You can" \
165                  " override this by setting the $%s environment variable."
166        message = message % (self.short_name, self.base_dir, self.environ_var)
167        # Display the message with style in a box #
168        from plumbing.common import rich_panel_print
169        rich_panel_print(message, "Large Download")
170        # Show a progress bar a bit like wget #
171        download_from_url(self.url,
172                          destination = self.tarball,
173                          uncompress  = False,
174                          user_agent  = "crest4 v" + crest4.__version__,
175                          stream      = True,
176                          progress    = True,
177                          desc        = self.short_name)
178        # Uncompress #
179        self.tarball.untargz_to(self.base_dir)
180        # Remove tarball #
181        self.tarball.remove()
182        # Remove macOS attribute files that get bundled in the tar #
183        for path in self.base_dir.flat_files:
184            if path.name.startswith("._"): path.remove()
185        # Check it worked #
186        del self.downloaded
187        assert self.downloaded
188
189    #--------------------------- Specific Indexes ----------------------------#
190    @property_cached
191    def blast_db(self):
192        """
193        Return a `BLASTDb` object that can be used for the sequence
194        similarity search.
195        """
196        # Download the database if it has not been done already #
197        if not self.downloaded: self.download()
198        # Create the database object #
199        from seqsearch.search.blast import BLASTdb
200        db = BLASTdb(self.path, seq_type='nucl')
201        # Create the database with `mkblastdb` if it's not made already #
202        db.create_if_not_exists(verbose=True)
203        # Return #
204        return db
205
206    @property_cached
207    def vsearch_db(self):
208        """
209        Return a `VSEARCHdb` object that can be used for the sequence
210        similarity search.
211        """
212        # Download the database if it has not been done already #
213        if not self.downloaded: self.download()
214        # Create the database object #
215        from seqsearch.search.vsearch import VSEARCHdb
216        db = VSEARCHdb(self.path.replace_extension('udb'))
217        # Create the database with `vsearch` if it's not made already #
218        db.create_if_not_exists(verbose=True)
219        # Return #
220        return db
221
222    #--------------------------- Loading the tree ----------------------------#
223    @property_cached
224    def tree(self):
225        """
226        Using the `.tre` file, we return an N-ary tree in memory.
227        Every node is characterized by a number. For instance between
228        1 and 32477.
229        """
230        # Load the tree with ete3 #
231        from ete3 import Tree
232        return Tree(self.path.replace_extension('tre'), format=8)
233
234    @property_cached
235    def node_to_name(self):
236        """
237        Using the `.names` file, we return a dictionary linking node numbers to
238        a tuple of species names and a minimum similarity fraction like 0.97.
239        Example: 51 -> (Proteobacteria, 0.8)
240        """
241        # Get the path of the file #
242        path = self.path.replace_extension('names')
243        # Define how to process each line #
244        def parse_lines(lines):
245            for line in lines:
246                if line.startswith('#') or not line: continue
247                num, name, frac = line.strip().split(',')
248                yield num, (name, frac)
249        # Create a dictionary #
250        with open(path, 'rt') as handle: return dict(parse_lines(handle))
251
252    @property_cached
253    def acc_to_node(self):
254        """
255        Using the `.map` file, we return a dictionary linking accession strings
256        to node numbers.
257        Example: HQ191339 -> 28386
258        """
259        # Get the path of the file #
260        path = self.path.replace_extension('map')
261        # Define how to process each line #
262        def parse_lines(lines):
263            for line in lines:
264                if line.startswith('#') or not line: continue
265                num, name = line.strip().split(',')
266                yield name, num
267        # Create a dictionary #
268        with open(path, 'rt') as handle: return dict(parse_lines(handle))
269
270    #--------------------------- Extra information ---------------------------#
271    @property
272    def rank_names(self):
273        return ['Root',           # 0 (This is life itself)
274                'Genome',         # 1 (This is for instance 'mitochondria')
275                'Domain',         # 2 (This is Bacteria, Archaea or Eucarya)
276                'Superkingdom',   # 3
277                'Kingdom',        # 4 (This is also called Superphylum)
278                'Phylum',         # 5
279                'Class',          # 6
280                'Order',          # 7
281                'Family',         # 8
282                'Genus',          # 9
283                'Species',        # 10
284                'Strain',         # 11
285                'Strain',         # 12
286                'Strain',         # 13 (There is a problem with too many ranks
287                'Strain']         # 14  in some databases requiring this)
